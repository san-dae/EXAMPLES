Node* buildTreeHelper(vector<int>& nodes, int i) {
    if (i >= nodes.size()) return nullptr;

    Node* root = new Node(nodes[i]);
    root->left = buildTreeHelper(nodes, 2 * i + 1);
    root->right = buildTreeHelper(nodes, 2 * i + 2);

    return root;
}
